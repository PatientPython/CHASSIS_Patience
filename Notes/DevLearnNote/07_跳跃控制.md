# 第七章 跳跃控制子状态机

## 7.1 概述

跳跃功能是平衡步兵底盘的高级运动技能，需要在短时间内协调多个控制目标：
1. **下蹲蓄力**: 收缩腿部储存势能
2. **爆发起跳**: 快速伸腿获得地面反作用力
3. **空中收腿**: 离地后收缩腿部避免碰撞
4. **缓冲着陆**: 触地瞬间吸收冲击

整个过程通过**六阶段子状态机**实现。

## 7.2 跳跃阶段定义

### 7.2.1 阶段枚举

```c
// Global/GlobalDeclare_Chassis.h
typedef enum {
    CH_JumpPhase_Wait = 0,       // 等待阶段
    CH_JumpPhase_Compress,       // 压缩蓄力
    CH_JumpPhase_Takeoff,        // 起跳伸展
    CH_JumpPhase_Retract,        // 收腿阶段
    CH_JumpPhase_AirFree,        // 空中自由
    CH_JumpPhase_Landing,        // 着陆缓冲
} ChassisJumpPhase_EnumTypeDef;
```

### 7.2.2 各阶段说明

| 阶段 | 目标 | 持续时间 | 退出条件 |
|------|------|----------|----------|
| Wait | 进入跳跃后的短暂准备 | ~50ms | 时间到达 |
| Compress | 下蹲蓄力，压缩腿部 | ~200ms | 腿长接近压缩目标 |
| Takeoff | 快速伸腿起跳 | ~100ms | 腿长达到阈值或离地 |
| Retract | 离地后快速收腿 | ~150ms | 收腿完成 |
| AirFree | 空中保持姿态 | 直到落地 | 触地检测 |
| Landing | 着陆缓冲 | ~200ms | 时间到达 |

## 7.3 跳跃状态机实现

### 7.3.1 主处理函数

```c
// Application/Chassis/Chassis_Stratgy.c L812-1018
void ChModeControl_JumpMode_RCControl(void) {
    // 获取当前时间
    uint32_t TimeNow = RunTimeGet();
    uint32_t ModeStartTime = GSTCH_Data.ST_ModeStartTime.RC_Jump;

    // 获取腿长平均值
    float LegLenAvg = (GSTCH_Data.LegLen1FB + GSTCH_Data.LegLen2FB) / 2.0f;
    float LegLen1Err = fabs(GSTCH_Data.LegLen1FB - LegLenJumpCompressTarget);
    float LegLen2Err = fabs(GSTCH_Data.LegLen2FB - LegLenJumpCompressTarget);

    // 离地标志
    bool IsOffGround = GSTCH_Data.F_OffGround1 || GSTCH_Data.F_OffGround2;

    // ========== 前置处理：进入跳跃后的短暂等待期 ==========
    if (TimeNow - ModeStartTime < CHMode_AllMode_PreProcessTime) {
        JumpPhase = CH_JumpPhase_Wait;
    }

    // ========== 状态机主逻辑 ==========
    switch (JumpPhase) {
        // ... 各阶段处理
    }

    // ========== 更新全局标志位 ==========
    GSTCH_Data.F_JumpTakeoff = (JumpPhase == CH_JumpPhase_Takeoff);
    GSTCH_Data.F_JumpRetract = (JumpPhase == CH_JumpPhase_Retract);
    GSTCH_Data.F_JumpLanding = (JumpPhase == CH_JumpPhase_Landing);

    // ========== 调用运动处理函数 ==========
    CH_MotionUpdateAndProcess(GST_RMCtrl);
}
```

## 7.4 各阶段详细实现

### 7.4.1 Wait - 等待阶段

**目标**: 进入跳跃后的短暂准备，稳定系统状态。

```c
case CH_JumpPhase_Wait: {
    // 等待期结束 → Compress
    if (TimeNow - ModeStartTime >= CHMode_AllMode_PreProcessTime) {
        JumpPhase = CH_JumpPhase_Compress;
    }

    // 保持低腿长，前馈力为重力补偿
    PID_SetKpKiKd(&GstCH_LegLen1PID, PID_LegLen_KpNorm, 0.0f, PID_LegLen_KdNorm);
    PID_SetKpKiKd(&GstCH_LegLen2PID, PID_LegLen_KpNorm, 0.0f, PID_LegLen_KdNorm);

    TD_Setr(&GstCH_LegLen1TD, TD_LegLen_rNorm);
    TD_Setr(&GstCH_LegLen2TD, TD_LegLen_rNorm);

    GST_RMCtrl.STCH_Default.LegLen1Des = LegLenLow;
    GST_RMCtrl.STCH_Default.LegLen2Des = LegLenLow;
    GST_RMCtrl.STCH_Default.Leg1FFForce = LegFForce_Gravity_1;
    GST_RMCtrl.STCH_Default.Leg2FFForce = LegFForce_Gravity_2;

    Chassis_DisFBClear();  // 清除位移反馈
    break;
}
```

**参数设置**:
| 参数 | 值 | 说明 |
|------|-----|------|
| 腿长目标 | `LegLenLow` (~0.18m) | 低腿长准备 |
| PID | 正常模式参数 | 稳定保持 |
| 前馈力 | 重力补偿 | 平衡自重 |

### 7.4.2 Compress - 压缩蓄力阶段

**目标**: 下蹲到压缩腿长，储存势能为起跳做准备。

```c
case CH_JumpPhase_Compress: {
    // 切换条件：双腿都接近压缩目标
    bool Leg1Compressed = (LegLen1Err < 0.025f) ||
                          (GSTCH_Data.LegLen1FB < LegLenJumpCompressTarget);
    bool Leg2Compressed = (LegLen2Err < 0.025f) ||
                          (GSTCH_Data.LegLen2FB < LegLenJumpCompressTarget);

    if (Leg1Compressed && Leg2Compressed) {
        JumpPhase = CH_JumpPhase_Takeoff;
    }

    // 压缩阶段 PID：较大增益快速响应
    PID_SetKpKiKd(&GstCH_LegLen1PID,
                  PID_LegLen_KpJump_Compress,    // 1000.0
                  0.0f,
                  PID_LegLen_KdJump_Compress);   // 108000.0
    PID_SetKpKiKd(&GstCH_LegLen2PID,
                  PID_LegLen_KpJump_Compress,
                  0.0f,
                  PID_LegLen_KdJump_Compress);

    TD_Setr(&GstCH_LegLen1TD, TD_LegLen_rNorm);
    TD_Setr(&GstCH_LegLen2TD, TD_LegLen_rNorm);

    GST_RMCtrl.STCH_Default.LegLen1Des = LegLenJumpCompressTarget;  // ~0.11m
    GST_RMCtrl.STCH_Default.LegLen2Des = LegLenJumpCompressTarget;
    GST_RMCtrl.STCH_Default.Leg1FFForce = LegFForce_Gravity_1;
    GST_RMCtrl.STCH_Default.Leg2FFForce = LegFForce_Gravity_2;
    break;
}
```

**参数设置**:
| 参数 | 值 | 说明 |
|------|-----|------|
| 腿长目标 | `LegLenJumpCompressTarget` (~0.11m) | 压缩腿长 |
| Kp | 1000.0 | 中等增益，平稳下蹲 |
| Kd | 108000.0 | 较大阻尼，防止过冲 |
| 前馈力 | 重力补偿 | 平衡自重 |

### 7.4.3 Takeoff - 起跳伸展阶段

**目标**: 快速伸腿，获得地面反作用力使机器人离地。

```c
case CH_JumpPhase_Takeoff: {
    // 切换条件：腿长达到阈值 或 检测到离地
    if ((LegLenAvg >= LegLenJumpRetractThreshold) || IsOffGround) {
        JumpPhase = CH_JumpPhase_Retract;
    }

    // 分段 PID 控制
    float TakeoffErr = LegLenJumpTarget - LegLenAvg;
    if (TakeoffErr > 0.030f) {
        // 远离目标：大 PID + 大前馈（快速伸展）
        PID_SetKpKiKd(&GstCH_LegLen1PID,
                      PID_LegLen_KpJump_Takeoff,    // 1800.0
                      0.0f,
                      PID_LegLen_KdJump_Takeoff);   // 0.0 (零阻尼)
        PID_SetKpKiKd(&GstCH_LegLen2PID,
                      PID_LegLen_KpJump_Takeoff,
                      0.0f,
                      PID_LegLen_KdJump_Takeoff);
    } else {
        // 接近目标：减小前馈（缓冲）
        PID_SetKpKiKd(&GstCH_LegLen1PID,
                      PID_LegLen_KpJump_Takeoff * 0.8f,
                      0.0f,
                      PID_LegLen_KdJump_Takeoff * 0.7f);
        PID_SetKpKiKd(&GstCH_LegLen2PID,
                      PID_LegLen_KpJump_Takeoff * 0.8f,
                      0.0f,
                      PID_LegLen_KdJump_Takeoff * 0.7f);
    }

    // 起跳阶段不加 TD 跟踪，追求快速响应
    GST_RMCtrl.STCH_Default.LegLen1Des = LegLenJumpTarget;  // ~0.4m
    GST_RMCtrl.STCH_Default.LegLen2Des = LegLenJumpTarget;
    GST_RMCtrl.STCH_Default.Leg1FFForce = LegFForce_Jump;   // 1000N
    GST_RMCtrl.STCH_Default.Leg2FFForce = LegFForce_Jump;
    break;
}
```

**关键设计**:
1. **零阻尼 (Kd=0)**: 不使用微分项，允许最大加速度
2. **大前馈力**: `LegFForce_Jump = 1000N` 提供爆发性输出
3. **分段控制**: 接近目标时减小增益，防止过度伸展
4. **退出条件**: 腿长达到阈值或离地检测触发

**参数设置**:
| 参数 | 值 | 说明 |
|------|-----|------|
| 腿长目标 | `LegLenJumpTarget` (~0.4m) | 完全伸展 |
| Kp (远) | 1800.0 | 高增益快速响应 |
| Kd (远) | 0.0 | 零阻尼，爆发输出 |
| Kp (近) | 1800.0 * 0.8 | 减小增益缓冲 |
| 前馈力 | `LegFForce_Jump` (1000N) | 爆发起跳 |

### 7.4.4 Retract - 收腿阶段

**目标**: 离地后快速收缩腿部，避免与地面碰撞，为空中姿态做准备。

```c
case CH_JumpPhase_Retract: {
    // 收腿完成判断
    float RetractErr = fabs(LegLenAvg - LegLenJumpRetractTarget);
    bool RetractDone = (RetractErr < 0.020f);

    // 切换条件：收腿完成 + (在空中 → AirFree / 已触地 → Landing)
    if (RetractDone) {
        if (IsOffGround) {
            JumpPhase = CH_JumpPhase_AirFree;
        } else {
            JumpPhase = CH_JumpPhase_Landing;
        }
    }

    // 分段 PID 控制
    float ErrThreshold = 0.030f;
    if (RetractErr > ErrThreshold) {
        // 远离目标：大 PID + 大前馈（快速收缩）
        PID_SetKpKiKd(&GstCH_LegLen1PID,
                      PID_LegLen_KpJump_Retract,    // 3000.0 (最高)
                      0.0f,
                      PID_LegLen_KdJump_Retract);   // 120000.0
        PID_SetKpKiKd(&GstCH_LegLen2PID,
                      PID_LegLen_KpJump_Retract,
                      0.0f,
                      PID_LegLen_KdJump_Retract);
    } else {
        // 接近目标：减小参数
        PID_SetKpKiKd(&GstCH_LegLen1PID,
                      PID_LegLen_KpJump_Retract * 0.8f,
                      0.0f,
                      PID_LegLen_KdJump_Retract * 0.8f);
        PID_SetKpKiKd(&GstCH_LegLen2PID,
                      PID_LegLen_KpJump_Retract * 0.8f,
                      0.0f,
                      PID_LegLen_KdJump_Retract * 0.8f);
    }

    TD_Setr(&GstCH_LegLen1TD, TD_LegLen_rNorm);
    TD_Setr(&GstCH_LegLen2TD, TD_LegLen_rNorm);

    GST_RMCtrl.STCH_Default.LegLen1Des = LegLenJumpRetractTarget;  // ~0.18m
    GST_RMCtrl.STCH_Default.LegLen2Des = LegLenJumpRetractTarget;
    // 负前馈：协助收缩
    GST_RMCtrl.STCH_Default.Leg1FFForce = -LegFForce_Gravity_1;
    GST_RMCtrl.STCH_Default.Leg2FFForce = -LegFForce_Gravity_2;
    break;
}
```

**关键设计**:
1. **最高 Kp**: `3000.0` 确保最快收缩速度
2. **负前馈**: `-LegFForce_Gravity` 协助腿部收缩
3. **分支退出**: 根据是否离地决定进入 AirFree 还是 Landing

**参数设置**:
| 参数 | 值 | 说明 |
|------|-----|------|
| 腿长目标 | `LegLenJumpRetractTarget` (~0.18m) | 收腿长度 |
| Kp (远) | 3000.0 | 最高增益，快速收缩 |
| Kp (近) | 3000.0 * 0.8 | 减小增益缓冲 |
| 前馈力 | `-LegFForce_Gravity` | 负重力补偿，协助收缩 |

### 7.4.5 AirFree - 空中自由阶段

**目标**: 在空中保持收缩姿态，等待落地。

```c
case CH_JumpPhase_AirFree: {
    // 切换条件：检测到触地
    if (!IsOffGround) {
        JumpPhase = CH_JumpPhase_Landing;
    }

    // 空中阶段：较小 PID 保持柔顺
    PID_SetKpKiKd(&GstCH_LegLen1PID,
                  PID_LegLen_KpJump_AirFree,    // 750.0 (较低)
                  0.0f,
                  PID_LegLen_KdJump_AirFree);   // 120000.0
    PID_SetKpKiKd(&GstCH_LegLen2PID,
                  PID_LegLen_KpJump_AirFree,
                  0.0f,
                  PID_LegLen_KdJump_AirFree);

    TD_Setr(&GstCH_LegLen1TD, TD_LegLen_rNorm);
    TD_Setr(&GstCH_LegLen2TD, TD_LegLen_rNorm);

    GST_RMCtrl.STCH_Default.LegLen1Des = LegLenJumpRetractTarget;
    GST_RMCtrl.STCH_Default.LegLen2Des = LegLenJumpRetractTarget;
    // 较小前馈，仅维持姿态
    GST_RMCtrl.STCH_Default.Leg1FFForce = LegFForce_Gravity_1 * 0.5f;
    GST_RMCtrl.STCH_Default.Leg2FFForce = LegFForce_Gravity_2 * 0.5f;
    break;
}
```

**参数设置**:
| 参数 | 值 | 说明 |
|------|-----|------|
| 腿长目标 | `LegLenJumpRetractTarget` | 保持收腿 |
| Kp | 750.0 | 较低增益，柔顺保持 |
| 前馈力 | `0.5 * LegFForce_Gravity` | 半重力补偿 |

### 7.4.6 Landing - 着陆缓冲阶段

case CH_JumpPhase_Landing: {

## 7.8 调试建议

### 7.8.1 观测变量

| 变量 | 含义 | 正常范围 |
|------|------|----------|
| `JumpPhase` | 当前跳跃阶段 | 0-5 |
| `GSTCH_Data.F_JumpTakeoff` | 起跳标志 | true/false |
| `GSTCH_Data.F_JumpRetract` | 收腿标志 | true/false |
| `GSTCH_Data.F_JumpLanding` | 着陆标志 | true/false |
| `GSTCH_Data.F_OffGround1/2` | 离地标志 | true/false |
| `LegLenAvg` | 平均腿长 | 0.11~0.40 m |

### 7.8.2 跳跃高度调整

**增加跳跃高度**:
1. 增大 `LegFForce_Jump`（跳跃前馈力）
2. 增大 `PID_LegLen_KpJump_Takeoff`（起跳 Kp）
3. 增大 `LegLenJumpTarget`（伸展目标腿长）

**减小跳跃高度**:
1. 减小 `LegFForce_Jump`
2. 减小 `PID_LegLen_KpJump_Takeoff`
3. 减小 `LegLenJumpTarget`

### 7.8.3 常见问题

**问题 1**: 跳不起来

**原因**: 前馈力不足或起跳速度太慢
**解决**:
- 增大 `LegFForce_Jump` 从 1000N 到 1500N
- 增大 `PID_LegLen_KpJump_Takeoff`
- 保持 `Kd = 0`（零阻尼）

**问题 2**: 落地摔倒

**原因**: 着陆缓冲不足
**解决**:
- 增大 `PID_LegLen_KdJump_Landing`
- 延长 Landing 阶段持续时间

**问题 3**: 收腿碰撞地面

**原因**: 收腿速度太慢
**解决**:
- 增大 `PID_LegLen_KpJump_Retract`
- 确保离地检测灵敏
